 الخطوة 1: رفع الموقع مجاناً على Netlify (3 دقائق)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح المتصفح: https://app.netlify.com/drop
2. سجّل دخول (Google أو GitHub - مجاني 100%)
3. في جهازك، افتح المجلد اللي فيه المشروع
4. اسحب مجلد **dist/** (مجلد واحد فقط) وأفلته في صفحة Netlify
5. ✅ الموقع الآن على الإنترنت!

ستحصل على رابط مثل:
  https://romantic-pasteur-123456.netlify.app

6. لتغيير الاسم (اختياري):
   → اذهب إلى: Site settings → Change site name
   → اكتب: royal-pipes-yemen
   → سيصبح: https://royal-pipes-yemen.netlify.app

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 2: جرب لوحة التحكم
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح: https://رابط-موقعك.netlify.app/admin
2. ستظهر بيانات الدخول في صندوق أصفر بالأعلى
3. اضغط زر **"نسخ"** لنسخ البيانات
4. الصقها في حقول الدخول
5. اضغط **"دخول لوحة التحكم"**

البيانات:
  اسم المستخدم: admin
  كلمة المرور:  royal2026

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 3: إضافة منتج جديد
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

من لوحة التحكم:
1. اذهب إلى "المنتجات"
2. اضغط "إضافة منتج"
3. أدخل:
   • اسم المنتج: ماسورة PVC 50 ملم
   • الفئة: أنظمة السباكة
   • التصنيف: مواسير PVC-U
   • السعر: 500
   • العملة: ريال
   • الوصف: ماسورة PVC-U أبيض، ضغط 10 بار
   • رابط الصورة: https://... (رابط صورة من الإنترنت)
4. اضغط "حفظ"
5. ✅ المنتج يظهر فوراً لجميع العملاء!
